			private const int WM_CLOSE = 16;

			private AutoClosingMessageBox(string text, string caption, int timeout)
			{
				this._caption = caption;
				this._timeoutTimer = new System.Threading.Timer(new TimerCallback(this.OnTimerElapsed), null, timeout, -1);
				using (this._timeoutTimer)
				{
					MessageBox.Show(text, caption);
				}
			}

			public static void Show(string text, string caption, int timeout)
			{
				new MainForm.AutoClosingMessageBox(text, caption, timeout);
			}

			private void OnTimerElapsed(object state)
			{
				IntPtr mbWnd = MainForm.AutoClosingMessageBox.FindWindow("#32770", this._caption);
				bool flag = mbWnd != IntPtr.Zero;
				if (flag)
				{
					MainForm.AutoClosingMessageBox.SendMessage(mbWnd, 16u, IntPtr.Zero, IntPtr.Zero);
				}
				this._timeoutTimer.Dispose();
			}

			[DllImport("user32.dll", SetLastError = true)]
			private static extern IntPtr FindWindow(string lpClassName, string lpWindowName);

			[DllImport("user32.dll", CharSet = CharSet.Auto)]
			private static extern IntPtr SendMessage(IntPtr hWnd, uint Msg, IntPtr wParam, IntPtr lParam);
		}

		private void performLogin()
		{
			MessageForm messageForm = null;
			try
			{
				this.loggedin = false;
				this.resetScreen();
				messageForm = new MessageForm("Logging System User in", Resources.SmallLogo);
				messageForm.Show();
				this.validateValues(new TextBox[]
				{
					this.usernameTextbox,
					this.passwordTextbox
				}, new string[]
				{
					"Username needs a value ! ",
					"Password needs a value !"
				});
				Login login = new Login();
				login.username = this.usernameTextbox.Text;
				login.password = this.passwordTextbox.Text;
				WebAPIResponse webAPIResponse = RESTManager.Instance.CallGenericRestPostBearerToken<string>("authentication", Settings.Default.SecurityURL, login.ToJSON(), null, this.securityTokenTextBox.Text);
				string jsonResult = webAPIResponse.ResponseResult;
				bool flag = webAPIResponse.ResponseCode == 200;
				if (!flag)
				{
					throw new Exception(string.Concat(new string[]
					{
						"Error occured. Response code : ",
						webAPIResponse.ResponseCode.ToString(),
						", Reason : ",
						webAPIResponse.ResponseDescription,
						", Detail : -----------\n",
						webAPIResponse.ResponseResult,
						"-----------\n"
					}));
				}
				LoginResponse loginResponse = JsonConvert.DeserializeObject<LoginResponse>(Convert.ToString(jsonResult), new JsonSerializerSettings
				{
					TypeNameHandling = TypeNameHandling.Auto,
					NullValueHandling = NullValueHandling.Ignore
				});
				this.securityTokenTextBox.Text = loginResponse.token;
				this.loginResultText.Text = loginResponse.loginResponse;
				messageForm.Close();
				messageForm.Dispose();
				this.loggedin = true;
			}
			catch (Exception ex)
			{
				bool flag2 = messageForm != null;
				if (flag2)
				{
					messageForm.Close();
					messageForm.Dispose();
				}
				bool showErrorMessageDetail = Settings.Default.ShowErrorMessageDetail;
				if (showErrorMessageDetail)
				{
					MessageBox.Show(ex.ToString());
				}
				else
				{
					MessageBox.Show(ex.Message);
				}
				this.loggedin = false;
			}
		}

		public static string Base64Encode(string plainText)
		{
			byte[] plainTextBytes = Encoding.UTF8.GetBytes(plainText);

		}

		public void validateValues(TextBox[] Textboxes, string[] ErrorValidationMessages)
		{
			int counter = 0;
			try
			{
				bool flag = Textboxes.Length != ErrorValidationMessages.Length;
				if (flag)
				{
					throw new Exception("Textbox count and message count are not equal !!");
				}
				for (int i = 0; i < Textboxes.Length; i++)
				{
					TextBox textbox = Textboxes[i];
					bool flag2 = textbox.Text.Trim() == "";
					if (flag2)
					{
						textbox.Focus();
						throw new Exception(ErrorValidationMessages[counter]);
					}
				}
			}
			catch (Exception e)
			{
				throw e;
			}
		}

				WebAPIResponse webAPIResponse = RESTManager.Instance.CallGenericRestPostBearerToken<string>("bankstatements", Settings.Default.BankStatementsURL, bankStatementRequest.ToJSON(), null, this.securityTokenTextBox.Text);
				string jsonResult = webAPIResponse.ResponseResult;
				bool flag = webAPIResponse.ResponseCode == 200;
				if (!flag)
				{
					throw new Exception(string.Concat(new string[]
					{
						"Error occured. Response code : ",
						webAPIResponse.ResponseCode.ToString(),
						", Reason : ",
						webAPIResponse.ResponseDescription,
						", -----------\n Detail : ",
						webAPIResponse.ResponseResult,
						"-----------\n"
					}));
				}
				BankStatementResult bankStatementResult = JsonConvert.DeserializeObject<BankStatementResult>(Convert.ToString(jsonResult), new JsonSerializerSettings
				{
					TypeNameHandling = TypeNameHandling.Auto,
					NullValueHandling = NullValueHandling.Ignore
				});
				try
				{
					this.FindAndKillProcess("AcroRd32");
				}
				catch (Exception exx)
				{
					MessageBox.Show("Could not Close Adobe. Error is : " + exx.ToString());
				}
				string fileName = Environment.CurrentDirectory + "\\LatestStatement{0}.pdf";
				for (int i = 1; i < 4; i++)
				{
					File.Delete(string.Format(fileName, i));
				}
				try
				{
					bool flag2 = bankStatementResult.bankStatements == null;
					if (flag2)
					{
						throw new Exception("Bankstatement is empty !!");
					}
				}
				catch
				{
				}
				messageForm.setMessage("Bankstatement retreived. Saving and opening bankstatement.");
				int fileCounter = 1;
				foreach (BankStatement bankstatement in bankStatementResult.bankStatements)
				{
					File.WriteAllBytes(string.Format(fileName, fileCounter), MainForm.Base64DecodeToByteArray(bankstatement.bankStatement));
					Process.Start(string.Format(fileName, fileCounter));
					fileCounter++;
				}
				messageForm.Close();
				messageForm.Dispose();
				this.resetScreen();
			}
			catch (Exception ex)
			{
				bool flag3 = messageForm != null;
				if (flag3)
				{
					messageForm.Close();
					messageForm.Dispose();
				}
				bool showErrorMessageDetail = Settings.Default.ShowErrorMessageDetail;
				if (showErrorMessageDetail)
				{
					MessageBox.Show(ex.ToString());
				}
				else
				{
					MessageBox.Show(ex.Message);
				}
			}
		}

		public bool FindAndKillProcess(string name)
		{
			Process[] processes = Process.GetProcesses();
			bool result;
			for (int i = 0; i < processes.Length; i++)
			{
				Process clsProcess = processes[i];
				bool flag = clsProcess.ProcessName.Contains(name);
				if (flag)
				{
					clsProcess.Kill();
					result = true;
					return result;
				}
			}
			result = false;
			return result;
		}
